                states.append( (0, k, dp[0][k]) )
        for z in queries:
            ans = 0
            for (r, k, t) in states:
                if t > z:
                    continue
                if best_cycle_time[r] == INF:
                    total = k
                else:
                    total = k + best_cycle_profit[r] * ((z - t) // best_cycle_time[r])
                if total > ans:
                    ans = total
            print(ans)
